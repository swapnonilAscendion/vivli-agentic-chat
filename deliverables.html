import asyncio from pdf_scraper import process_pdf_with_scraping async def main(): result = await process_pdf_with_scraping( pdf_path="sample.pdf", max_depth=2, rate_limit=2.0, timeout=30 ) print(f"Processed {result['scraping_results']['total_urls_processed']} URLs") print(f"Success rate: {result['scraping_results']['success_rate']}") combined_text = result['combined_data']['all_text'] asyncio.run(main())
